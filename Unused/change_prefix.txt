def FC(id,prefixes):
    for prefix in prefixes:
        if prefix['id']==id:
            return prefix['prefix']
    
    with open("prefixes.json", "w") as fileout:
        prefixes.append({"id":id, "prefix":"!"})
        json.dump(prefixes,fileout)
        return "!"
